### Descriptor Protocol — The Foundation of Python's Type Machinery

```python
from __future__ import annotations
from typing import Any, Generic, TypeVar, overload
from weakref import WeakKeyDictionary

T = TypeVar("T")
V = TypeVar("V")


class Validated(Generic[V]):
    """Descriptor enforcing runtime validation with zero per-access overhead.

    CASCADE AWARENESS:
    - PY3: Generic[V] enables mypy inference at assignment sites
    - PY5: WeakKeyDictionary prevents test fixture leaks across parametrize runs
    - PY6: __set_name__ avoids per-access string formatting (hot-path safe)
    - PY8: Structured error includes field name for observability correlation
    """

    def __init__(self, validator: type[V] | callable[[Any], V]) -> None:
        self._validator = validator
        self._values: WeakKeyDictionary[Any, V] = WeakKeyDictionary()
        self._name: str = ""

    def __set_name__(self, owner: type, name: str) -> None:
        self._name = name

    @overload
    def __get__(self, obj: None, objtype: type) -> Validated[V]: ...
    @overload
    def __get__(self, obj: Any, objtype: type) -> V: ...

    def __get__(self, obj: Any, objtype: type) -> Validated[V] | V:
        if obj is None:
            return self
        try:
            return self._values[obj]
        except KeyError:
            raise AttributeError(
                f"{objtype.__name__}.{self._name} not set"
            ) from None

    def __set__(self, obj: Any, value: Any) -> None:
        try:
            self._values[obj] = self._validator(value)
        except (TypeError, ValueError) as exc:
            raise ValueError(
                f"Validation failed for {type(obj).__name__}.{self._name}: {exc}"
            ) from exc
```

### Generator-Based Resource Pipeline

```python
from collections.abc import Generator, Iterable
from contextlib import contextmanager
from pathlib import Path
import mmap


@contextmanager
def mapped_file(path: Path) -> Generator[mmap.mmap]:
    """Memory-mapped file access with guaranteed cleanup.

    CASCADE: PY4 async version uses async contextmanager.
    PY5 fixtures wrap this for test isolation. PY6 profiles show
    mmap outperforms read() above 10MB.
    """
    f = open(path, "rb")  # noqa: SIM115
    try:
        mm = mmap.mmap(f.fileno(), 0, access=mmap.ACCESS_READ)
        try:
            yield mm
        finally:
            mm.close()
    finally:
        f.close()


def chunked_iter(iterable: Iterable[T], size: int) -> Generator[list[T]]:
    """Yield fixed-size chunks without materializing the full sequence."""
    chunk: list[T] = []
    for item in iterable:
        chunk.append(item)
        if len(chunk) == size:
            yield chunk
            chunk = []
    if chunk:
        yield chunk
```

### Protocol-Based Strategy — Replacing Abstract Classes

```python
from typing import Protocol, runtime_checkable
from decimal import Decimal
from dataclasses import dataclass


@runtime_checkable
class PricingStrategy(Protocol):
    """Protocol over ABC: testable without inheritance, composable with DI.

    CASCADE:
    - PY3: runtime_checkable enables isinstance() guards at API boundaries
    - PY5: Any callable matching signature works as test double — no mock needed
    - PY7: FastAPI Depends() resolves this via the DI container
    - PY8: Strategy identity logged as span attribute for pricing audit trail
    """

    def calculate(self, base_price: Decimal, quantity: int) -> Decimal: ...


@dataclass(frozen=True, slots=True)
class TieredPricing:
    """Concrete strategy — frozen dataclass for hashability and thread safety."""

    tiers: tuple[tuple[int, Decimal], ...]

    def calculate(self, base_price: Decimal, quantity: int) -> Decimal:
        discount = Decimal("0")
        for threshold, rate in self.tiers:
            if quantity >= threshold:
                discount = rate
        return base_price * quantity * (1 - discount)


# Strategy as plain function — equally valid, zero ceremony
def flat_discount(rate: Decimal) -> PricingStrategy:
    """Closure-based strategy: captures rate, satisfies Protocol."""

    def calculate(base_price: Decimal, quantity: int) -> Decimal:
        return base_price * quantity * (1 - rate)

    # Attach to a namespace for Protocol compliance
    class _Flat:
        calculate = staticmethod(calculate)

    return _Flat()
```

### Dependency Injection Without a Framework

```python
from typing import TypeVar, Generic
from collections.abc import Callable
from contextlib import asynccontextmanager
import asyncio

S = TypeVar("S")


class Container:
    """Minimal DI container — 40 lines replacing inject/dependency-injector.

    CASCADE:
    - PY4: Supports async factories via asynccontextmanager
    - PY5: Override in tests with container.register(Service, mock_factory)
    - PY7: FastAPI Depends() delegates to this for non-request-scoped deps
    """

    def __init__(self) -> None:
        self._factories: dict[type, Callable] = {}
        self._singletons: dict[type, object] = {}

    def register(
        self, interface: type[S], factory: Callable[..., S], *, singleton: bool = False
    ) -> None:
        self._factories[interface] = (factory, singleton)

    def resolve(self, interface: type[S]) -> S:
        factory, singleton = self._factories[interface]
        if singleton:
            if interface not in self._singletons:
                self._singletons[interface] = factory()
            return self._singletons[interface]  # type: ignore[return-value]
        return factory()

    def reset(self) -> None:
        """Clear singletons — essential for test isolation (PY5)."""
        self._singletons.clear()
```

### TypeGuard for Runtime Narrowing

```python
from typing import TypeGuard, Any


def is_non_empty_str(value: Any) -> TypeGuard[str]:
    """Runtime narrowing that mypy trusts.

    CASCADE: PY7 uses this in FastAPI validators, PY8 in log sanitization.

    return isinstance(value, str) and len(value) > 0


def is_positive_int(value: Any) -> TypeGuard[int]:
```

## PY4: Async Reactor — Structured Concurrency

> **Sources:** `async-python-patterns` × `python-error-handling` × `python-pro`  
> **Interconnection:** Async choices constrain PY2 patterns (no sync Observer), PY5 test runners, PY6 profiling tools.

Structured concurrency is Python 3.11+'s `TaskGroup` — every spawned task has a
defined lifetime, cancellation propagates, and exceptions never silently vanish.

### TaskGroup with Typed Results and Error Handling

```python
import asyncio
from collections.abc import Awaitable, Callable, Sequence
from dataclasses import dataclass
from typing import TypeVar, Generic

R = TypeVar("R")


@dataclass(frozen=True)
class TaskResult(Generic[R]):
    """Typed result container with error capture — no silent failures.

    CASCADE:
    - PY3: Generic[R] preserves return type through gather operations
    - PY5: Test assertions on .error enable precise failure checking
    - PY8: Failed tasks emit structured error spans automatically
    """

    value: R | None = None
    error: BaseException | None = None

    @property
    def ok(self) -> bool:
        return self.error is None


async def gather_typed(
    *coros: Awaitable[R],
    max_concurrency: int | None = None,
) -> list[TaskResult[R]]:
    """Bounded concurrent gather with per-task error isolation.

    Unlike asyncio.gather(return_exceptions=True), this preserves types
    and never mixes exceptions with return values.
    """
    sem = asyncio.Semaphore(max_concurrency) if max_concurrency else None
    results: list[TaskResult[R]] = []

    async def _run(coro: Awaitable[R], idx: int) -> None:
        if sem:
            async with sem:
                await _execute(coro, idx)
        else:
            await _execute(coro, idx)

    async def _execute(coro: Awaitable[R], idx: int) -> None:
        try:
            value = await coro
            results[idx] = TaskResult(value=value)
        except Exception as exc:
            results[idx] = TaskResult(error=exc)

    results = [TaskResult[R]()] * len(coros)
    async with asyncio.TaskGroup() as tg:
        for i, coro in enumerate(coros):
            tg.create_task(_run(coro, i))

    return results
```

### Async Context Manager with Lifecycle Hooks

```python
from contextlib import asynccontextmanager
from collections.abc import AsyncGenerator
from typing import Any
import asyncio
import logging

logger = logging.getLogger(__name__)


@asynccontextmanager
async def managed_pool(
    factory: Callable[[], Awaitable[Any]],
    size: int = 10,
    health_interval: float = 30.0,
) -> AsyncGenerator[asyncio.Queue]:
    """Connection pool with health checks and structured shutdown.

    CASCADE:
    - PY2: Factory pattern for pluggable backends (DB, HTTP, gRPC)
    - PY5: Inject mock factory in tests — pool lifecycle still exercised
    - PY6: size tuning based on profiler's connection wait histogram
    - PY8: Health check failures emit WARNING spans with pool stats
    """
    pool: asyncio.Queue = asyncio.Queue(maxsize=size)
    health_task: asyncio.Task | None = None

    # Fill pool
    for _ in range(size):
        conn = await factory()
        await pool.put(conn)

    async def _health_loop() -> None:
        while True:
            await asyncio.sleep(health_interval)
            checked = 0
            for _ in range(pool.qsize()):
                conn = await pool.get()
                try:
                    await conn.ping()  # type: ignore[attr-defined]
                    await pool.put(conn)
                except Exception:
                    logger.warning("Replacing dead connection in pool")
                    new_conn = await factory()
                    await pool.put(new_conn)
                checked += 1

    try:
        health_task = asyncio.create_task(_health_loop())
        yield pool
    finally:
        if health_task:
            health_task.cancel()
            with contextlib.suppress(asyncio.CancelledError):
                await health_task
        # Drain and close
        while not pool.empty():
            conn = await pool.get()
            with contextlib.suppress(Exception):
                await conn.close()  # type: ignore[attr-defined]
```

### Property-Based Testing with Hypothesis

```python
from hypothesis import given, strategies as st, settings, HealthCheck
from hypothesis.stateful import RuleBasedStateMachine, rule, invariant
from decimal import Decimal


class PricingStateMachine(RuleBasedStateMachine):
```

### Profiling-Driven Caching with Metrics

```python
from functools import lru_cache, wraps
from typing import TypeVar, ParamSpec, Callable
from time import perf_counter_ns
import logging

P = ParamSpec("P")
R = TypeVar("R")

logger = logging.getLogger(__name__)


def profiled_cache(
    maxsize: int = 256,
    *,
    warn_ms: float = 5.0,
    metric_name: str = "",
) -> Callable[[Callable[P, R]], Callable[P, R]]:
    """LRU cache with hit/miss metrics and slow-call warnings.

    CASCADE:
    - PY5: Tests verify cache behavior via .cache_info() assertions
    - PY8: Metrics emitted as OpenTelemetry histogram — no separate instrumentation
    """

    def decorator(fn: Callable[P, R]) -> Callable[P, R]:
        cached_fn = lru_cache(maxsize=maxsize)(fn)
        name = metric_name or f"{fn.__module__}.{fn.__qualname__}"

        @wraps(fn)
        def wrapper(*args: P.args, **kwargs: P.kwargs) -> R:
            start = perf_counter_ns()
            result = cached_fn(*args, **kwargs)
            elapsed_ms = (perf_counter_ns() - start) / 1_000_000

            info = cached_fn.cache_info()
            if elapsed_ms > warn_ms:
                logger.warning(
                    "Slow call: %s took %.2fms (hits=%d, misses=%d)",
                    name, elapsed_ms, info.hits, info.misses,
                )
            return result

        wrapper.cache_info = cached_fn.cache_info  # type: ignore[attr-defined]
        wrapper.cache_clear = cached_fn.cache_clear  # type: ignore[attr-defined]
        return wrapper

    return decorator
```

### `__slots__` + Memory Layout for Data-Heavy Objects

### FastAPI with Full Type Propagation and DI

```python
from fastapi import FastAPI, Depends, HTTPException, status
from pydantic import BaseModel, Field, ConfigDict
from typing import Annotated
from uuid import UUID
from datetime import datetime


# --- Pydantic models: PY3 types propagate to OpenAPI schema ---

class UserCreate(BaseModel):
    model_config = ConfigDict(strict=True)

    name: str = Field(min_length=1, max_length=200)
    email: str = Field(pattern=r"^[\w.-]+@[\w.-]+\.\w+$")


class UserResponse(BaseModel):
    model_config = ConfigDict(from_attributes=True)

    id: UUID
    name: str
    email: str
    created_at: datetime


# --- DI wiring: PY2 Container bridges to FastAPI Depends ---

from .container import Container  # from PY2
from .repository import Repository  # from PY3
from .models import User

container = Container()


async def get_user_repo() -> Repository[User]:
    """FastAPI dependency that resolves from our DI container.

    CASCADE:
    - PY2: Container.resolve() called — singleton/factory semantics preserved
    - PY5: Override with app.dependency_overrides in integration tests
    - PY8: Dependency resolution traced as child span
    """
    return container.resolve(Repository[User])


UserRepo = Annotated[Repository[User], Depends(get_user_repo)]

app = FastAPI(title="APEX Example")


@app.post("/users", response_model=UserResponse, status_code=status.HTTP_201_CREATED)
async def create_user(body: UserCreate, repo: UserRepo) -> User:
    user = User(name=body.name, email=body.email)
    return await repo.save(user)


@app.get("/users/{user_id}", response_model=UserResponse)
async def get_user(user_id: UUID, repo: UserRepo) -> User:
    user = await repo.get(user_id)
    if user is None:
        raise HTTPException(status_code=404, detail="User not found")
    return user
```

### Structured Error Hierarchy with Observability

```python
from dataclasses import dataclass, field
from typing import Any
from enum import StrEnum, auto
import logging
import traceback


class ErrorSeverity(StrEnum):
    LOW = auto()
    MEDIUM = auto()
    HIGH = auto()
    CRITICAL = auto()


@dataclass(frozen=True)
class AppError(Exception):
    """Structured application error — machine-parseable, human-readable.

    CASCADE:
    - PY2: Hierarchy replaces magic strings — pattern-matched in handlers
    - PY3: frozen dataclass satisfies Exception + Hashable protocols
    - PY5: Assertions on .code and .severity — no string parsing in tests
    - PY7: FastAPI exception handler maps .severity to HTTP status
    - PY8: Structured fields become OpenTelemetry span attributes
    """

    message: str
    code: str = "UNKNOWN"
    severity: ErrorSeverity = ErrorSeverity.MEDIUM
    context: dict[str, Any] = field(default_factory=dict)
    cause: Exception | None = field(default=None, repr=False)

    def __str__(self) -> str:
        return f"[{self.code}] {self.message}"

    def to_dict(self) -> dict[str, Any]:
        return {
            "code": self.code,
            "message": self.message,
            "severity": self.severity.value,
            "context": self.context,
        }


class NotFoundError(AppError):
    def __init__(self, entity: str, entity_id: Any, **ctx: Any) -> None:
        super().__init__(
            message=f"{entity} {entity_id} not found",
            code="NOT_FOUND",
            severity=ErrorSeverity.LOW,
            context={"entity": entity, "entity_id": str(entity_id), **ctx},
        )


class ValidationError(AppError):
    def __init__(self, field: str, reason: str, **ctx: Any) -> None:
        super().__init__(
            message=f"Validation failed on {field}: {reason}",
            code="VALIDATION_ERROR",
            severity=ErrorSeverity.MEDIUM,
            context={"field": field, "reason": reason, **ctx},
        )
```

### OpenTelemetry Integration with Automatic Span Attributes

```python
from opentelemetry import trace
from opentelemetry.trace import StatusCode
from functools import wraps
from typing import ParamSpec, TypeVar, Callable
import logging

P = ParamSpec("P")
R = TypeVar("R")

tracer = trace.get_tracer("forge.python.apex")
logger = logging.getLogger(__name__)


def traced(
    span_name: str | None = None,
    *,
    record_args: bool = False,
) -> Callable[[Callable[P, R]], Callable[P, R]]:
    """Decorator adding OpenTelemetry tracing with structured attributes.

    CASCADE:
    - PY5: Noop in tests (tracer returns NoOpSpan) — zero test overhead
    - PY6: Span durations feed the performance profiling dashboard
    - PY7: FastAPI middleware auto-traces routes; this covers service layer
    """

    def decorator(fn: Callable[P, R]) -> Callable[P, R]:
        name = span_name or f"{fn.__module__}.{fn.__qualname__}"

        @wraps(fn)
        async def async_wrapper(*args: P.args, **kwargs: P.kwargs) -> R:
            with tracer.start_as_current_span(name) as span:
                if record_args and kwargs:
                    for k, v in kwargs.items():
                        span.set_attribute(f"arg.{k}", str(v))
                try:
                    result = await fn(*args, **kwargs)
                    span.set_status(StatusCode.OK)
                    return result
                except AppError as exc:
                    span.set_status(StatusCode.ERROR, str(exc))
                    span.set_attribute("error.code", exc.code)
                    span.set_attribute("error.severity", exc.severity.value)
                    raise
                except Exception as exc:
                    span.set_status(StatusCode.ERROR, str(exc))
                    span.record_exception(exc)
                    raise

        @wraps(fn)
        def sync_wrapper(*args: P.args, **kwargs: P.kwargs) -> R:
            with tracer.start_as_current_span(name) as span:
                try:
                    result = fn(*args, **kwargs)
                    span.set_status(StatusCode.OK)
                    return result
                except Exception as exc:
                    span.set_status(StatusCode.ERROR, str(exc))
                    span.record_exception(exc)
                    raise

        import asyncio
        if asyncio.iscoroutinefunction(fn):
            return async_wrapper  # type: ignore[return-value]
        return sync_wrapper

    return decorator
```

### Cascade 1: Adding a New Domain Entity

**Trigger:** "Add an `Order` entity to the system."

| Step | Module | Action | Cascade Effect |
|------|--------|--------|----------------|
| 1 | PY1 | Define `Order` with `__slots__`, frozen dataclass | Memory-efficient, immutable |
| 2 | PY3 | Add `OrderT = TypeVar("OrderT", bound=Order)` | Enables generic repository |
| 3 | PY2 | Register `Repository[Order]` in DI Container | Decoupled from implementation |
| 4 | PY4 | Implement `AsyncOrderRepo` with connection pool | Structured concurrency lifecycle |
| 5 | PY7 | Add FastAPI routes with `Depends(get_order_repo)` | Type-safe API, auto OpenAPI docs |
| 6 | PY5 | Create `InMemoryRepository[Order]` fixture | Tests exercise real async protocol |
| 7 | PY8 | Add `NotFoundError("Order", id)` + `@traced` | Structured errors + OTel spans |
| 8 | PY6 | Profile order listing, add `@profiled_cache` if hot | Data-driven optimization |

**Interconnection:** Removing any single step creates drift. PY7 without PY5 = untested API.
PY4 without PY8 = unobservable concurrency. PY3 without PY2 = types without boundaries.

### Cascade 2: Migrating Sync to Async

**Trigger:** "Profile shows I/O wait dominating response time."

| Step | Module | Action | Why This Order |
|------|--------|--------|----------------|
| 1 | PY6 | Profile → confirm I/O bound (not CPU) | Don't async a CPU problem |
| 2 | PY4 | Replace blocking calls with `await`, add TaskGroup | Structured concurrency |
| 3 | PY2 | Update DI to async factories | Sync singletons can't hold async resources |
| 4 | PY3 | Add `Awaitable[T]` return types, update Protocol | mypy catches sync/async mismatches |
| 5 | PY5 | Switch to `pytest-asyncio`, add `anyio` backend | Test runner must match runtime |
| 6 | PY7 | Verify FastAPI routes already async (they should be) | Framework alignment check |
| 7 | PY8 | Update `@traced` to detect coroutines automatically | Observability follows the change |

### Cascade 3: Hardening for Production

**Trigger:** "Moving from prototype to production deployment."

| Step | Module | Action | Why This Order |
