```python
from dataclasses import asdict, dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import Iterable
import hashlib
import json
import mimetypes
import os
import sqlite3

DRIVE_MAP = {
    "C": {"label": "NVMe SSD", "class": "primary"},
    "D": {"label": "USB", "class": "removable"},
    "F": {"label": "USB", "class": "removable"},
    "G": {"label": "USB", "class": "removable"},
    "I": {"label": "SD card", "class": "removable"},
    "J": {"label": "2TB USB exFAT", "class": "archive"},
}

LANE_RULES = [
    ("E", ("misconduct", "canon", "jtc", "mcneill", "ex parte")),
    ("D", ("ppo", "protection order", "2023-5907-pp")),
    ("F", ("appeal", "coa", "7.212", "366810")),
    ("C", ("1983", "federal", "due process", "monell")),
    ("A", ("custody", "parenting time", "722.23", "2024-001507-dc")),
    ("B", ("housing", "lease", "eviction", "2025-002760-cz")),
]

@dataclass(slots=True)
class UniverseFile:
    drive: str
    drive_label: str
    path: str
    relative_path: str
    mime_type: str
    size_bytes: int
    sha256_head: str
    discovered_utc: str
    lane_hint: str
    source_system: str
    tags: list[str] = field(default_factory=list)

class LitigationUniverseScanner:
    def __init__(self, roots: dict[str, str], db_path: str) -> None:
        self.roots = {k: Path(v) for k, v in roots.items()}
        self.db_path = Path(db_path)

    def scan(self) -> list[UniverseFile]:
        records: list[UniverseFile] = []
        for drive, root in self.roots.items():
            if not root.exists():
                continue
            records.extend(self._scan_drive(drive, root))
        self._persist(records)
        return records

    def _scan_drive(self, drive: str, root: Path) -> list[UniverseFile]:
        results: list[UniverseFile] = []
        for current_root, _, files in os.walk(root):
            for name in files:
                path = Path(current_root, name)
                mime_type = mimetypes.guess_type(path.name)[0] or "application/octet-stream"
                head_digest = self._hash_head(path)
                lane_hint = self._infer_lane(path)
                results.append(
                    UniverseFile(
                        drive=drive,
                        drive_label=DRIVE_MAP[drive]["label"],
                        path=str(path),
                        relative_path=str(path.relative_to(root)),
                        mime_type=mime_type,
                        size_bytes=path.stat().st_size,
                        sha256_head=head_digest,
                        discovered_utc=datetime.now(timezone.utc).isoformat(),
                        lane_hint=lane_hint,
                        source_system=DRIVE_MAP[drive]["class"],
                        tags=self._derive_tags(path, lane_hint),
                    )
                )
        return results

    def _hash_head(self, path: Path, chunk_size: int = 131072) -> str:
        digest = hashlib.sha256()
        with path.open("rb") as handle:
            digest.update(handle.read(chunk_size))
        return digest.hexdigest()

    def _infer_lane(self, path: Path) -> str:
        haystack = str(path).lower()
        for lane, markers in LANE_RULES:
            if any(marker in haystack for marker in markers):
                return lane
        return "A"

    def _derive_tags(self, path: Path, lane_hint: str) -> list[str]:
        suffix = path.suffix.lower()
        tags = [f"lane:{lane_hint}", f"ext:{suffix or 'none'}"]
        if "ex parte" in path.name.lower():
            tags.append("risk:ex-parte")
        if suffix in {".pdf", ".docx", ".txt"}:
            tags.append("ocr:eligible")
        return tags

    def _persist(self, records: Iterable[UniverseFile]) -> None:
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                """
                CREATE TABLE IF NOT EXISTS litigation_universe (
                    path TEXT PRIMARY KEY,
                    drive TEXT NOT NULL,
                    lane_hint TEXT NOT NULL,
                    payload_json TEXT NOT NULL
                )
                """
            )
            conn.executemany(
                """
                INSERT OR REPLACE INTO litigation_universe(path, drive, lane_hint, payload_json)
                VALUES(:path, :drive, :lane_hint, :payload_json)
                """,
                [
                    {
                        "path": record.path,
                        "drive": record.drive,
                        "lane_hint": record.lane_hint,
                        "payload_json": json.dumps(asdict(record), ensure_ascii=False),
                    }
                    for record in records
                ],
            )

scanner = LitigationUniverseScanner(
    roots={"C": r"C:\\Litigation", "D": r"D:\\Litigation", "J": r"J:\\Archive"},
    db_path=r"C:\\LitigationOS\\data\\litigation_universe.db",
)
discovered = scanner.scan()
print(f"discovered={len(discovered)}")
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Iterable
import hashlib
import json
import sqlite3

@dataclass(slots=True)
class OCRPage:
    page_number: int
    text: str
    confidence: float

@dataclass(slots=True)
class EvidenceAtom:
    source_path: str
    lane: str
    digest: str
    pages: list[OCRPage]
    bates_start: str
    bates_end: str
    authentication_basis: list[str] = field(default_factory=list)

class BatesAllocator:
    def __init__(self, prefix: str = "PIGORS") -> None:
        self.prefix = prefix
        self.counters: dict[str, int] = {}

    def allocate(self, lane: str, page_count: int) -> tuple[str, str]:
        start = self.counters.get(lane, 1)
        end = start + max(page_count, 1) - 1
        self.counters[lane] = end + 1
        return (
            f"{self.prefix}-{lane}-{start:06d}",
            f"{self.prefix}-{lane}-{end:06d}",
        )

class EvidenceFusionReactor:
    def __init__(self, db_path: str, allocator: BatesAllocator | None = None) -> None:
        self.db_path = Path(db_path)
        self.allocator = allocator or BatesAllocator()

    def fuse_document(self, source_path: str, lane: str, extracted_pages: Iterable[str]) -> EvidenceAtom:
        pages = [
            OCRPage(page_number=index, text=text.strip(), confidence=self._score(text))
            for index, text in enumerate(extracted_pages, start=1)
        ]
        digest = self._content_digest(pages)
        start, end = self.allocator.allocate(lane, len(pages))
        atom = EvidenceAtom(
            source_path=source_path,
            lane=lane,
            digest=digest,
            pages=pages,
            bates_start=start,
            bates_end=end,
            authentication_basis=self._authentication_basis(source_path, pages),
        )
        self._persist(atom)
        return atom

    def deduplicate(self, atom: EvidenceAtom) -> bool:
        with sqlite3.connect(self.db_path) as conn:
            row = conn.execute(
                "SELECT 1 FROM evidence_atoms WHERE digest = ?",
                (atom.digest,),
            ).fetchone()
        return row is not None

    def _content_digest(self, pages: list[OCRPage]) -> str:
        digest = hashlib.sha256()
        for page in pages:
            digest.update(page.text.encode("utf-8", errors="ignore"))
        return digest.hexdigest()

    def _score(self, text: str) -> float:
        token_count = max(len(text.split()), 1)
        uppercase_penalty = min(sum(1 for c in text if c.isupper()) / max(len(text), 1), 0.2)
        return round(max(0.35, min(0.99, (token_count / 250.0) + 0.45 - uppercase_penalty)), 3)

    def _authentication_basis(self, source_path: str, pages: list[OCRPage]) -> list[str]:
        basis = ["MRE 901(b)(1)", "MRE 902(11)"]
        suffix = Path(source_path).suffix.lower()
        if suffix in {".jpg", ".png", ".tif", ".tiff"}:
            basis.append("Image metadata + witness recognition")
        if any("text message" in page.text.lower() for page in pages):
            basis.append("Device-origin message export verification")
        return basis

    def _persist(self, atom: EvidenceAtom) -> None:
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        with sqlite3.connect(self.db_path) as conn:
            conn.execute(
                """
                CREATE TABLE IF NOT EXISTS evidence_atoms (
                    digest TEXT PRIMARY KEY,
                    lane TEXT NOT NULL,
                    source_path TEXT NOT NULL,
                    bates_start TEXT NOT NULL,
                    bates_end TEXT NOT NULL,
                    payload_json TEXT NOT NULL
                )
                """
            )
            conn.execute(
                """
                INSERT OR REPLACE INTO evidence_atoms
                (digest, lane, source_path, bates_start, bates_end, payload_json)
                VALUES (?, ?, ?, ?, ?, ?)
                """,
                (
                    atom.digest,
                    atom.lane,
                    atom.source_path,
                    atom.bates_start,
                    atom.bates_end,
                    json.dumps(
                        {
                            "digest": atom.digest,
                            "lane": atom.lane,
                            "source_path": atom.source_path,
                            "pages": [page.__dict__ for page in atom.pages],
                            "authentication_basis": atom.authentication_basis,
                        },
                        ensure_ascii=False,
                    ),
                ),
            )

reactor = EvidenceFusionReactor(db_path=r"C:\\LitigationOS\\data\\evidence_atoms.db")
atom = reactor.fuse_document(
    source_path=r"J:\\Archive\\custody\\messages-export.pdf",
    lane="A",
    extracted_pages=["Message thread regarding parenting time", "Follow-up exhibit summary"],
)
print(atom.bates_start, atom.bates_end, atom.authentication_basis)
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Iterable
import math

@dataclass(slots=True)
class Quote:
    quote_id: str
    speaker: str
    text: str
    source_ref: str
    date_key: str
    lane: str

@dataclass(slots=True)
class Contradiction:
    primary_quote_id: str
    impeaching_quote_id: str
    inconsistency_axis: str
    severity: float
    script: list[str] = field(default_factory=list)

class CredibilityScorer:
    def score(self, contradictions: Iterable[Contradiction], prior_findings: int) -> float:
        contradiction_weight = sum(item.severity for item in contradictions)
        damping = math.log(prior_findings + 1, 10) + 1
        return round(max(0.0, 100.0 - (contradiction_weight * 7.5 * damping)), 2)

class AdversaryDestructionMatrix:
    def __init__(self) -> None:
        self.scorer = CredibilityScorer()

    def detect(self, quotes: list[Quote]) -> list[Contradiction]:
        contradictions: list[Contradiction] = []
        for index, left in enumerate(quotes):
            for right in quotes[index + 1:]:
                if left.speaker != right.speaker:
                    continue
                axis = self._axis(left.text, right.text)
                if not axis:
                    continue
                severity = self._severity(left.text, right.text, axis)
                contradictions.append(
                    Contradiction(
                        primary_quote_id=left.quote_id,
                        impeaching_quote_id=right.quote_id,
                        inconsistency_axis=axis,
                        severity=severity,
                        script=self._script(left, right, axis),
                    )
                )
        return sorted(contradictions, key=lambda item: item.severity, reverse=True)

    def _axis(self, left: str, right: str) -> str | None:
        pair = f"{left.lower()} || {right.lower()}"
        if "never" in pair and "always" in pair:
            return "absolute-position-reversal"
        if "did not" in pair and "did" in pair:
            return "action-admission-conflict"
        if any(token in pair for token in ("safe", "unsafe", "threat", "protective")):
            return "safety-narrative-conflict"
        if any(token in pair for token in ("schedule", "exchange", "pickup", "drop off")):
            return "parenting-time-sequence-conflict"
        return None

    def _severity(self, left: str, right: str, axis: str) -> float:
        overlap = len(set(left.lower().split()) & set(right.lower().split()))
        base = {
            "absolute-position-reversal": 9.8,
            "action-admission-conflict": 9.4,
            "safety-narrative-conflict": 8.8,
            "parenting-time-sequence-conflict": 8.2,
        }[axis]
        return round(base + min(overlap / 20.0, 0.9), 2)

    def _script(self, left: Quote, right: Quote, axis: str) -> list[str]:
        return [
            f"COMMIT: You testified in {left.source_ref} that '{left.text[:90]}...'",
            f"PIN: That was your statement on {left.date_key}, correct?",
            f"CONFRONT: In {right.source_ref}, you also said '{right.text[:90]}...'",
            f"EXHIBIT: Display {left.source_ref} and {right.source_ref} under MRE 613 ({axis}).",
        ]

    def build_matrix(self, target: str, quotes: list[Quote], prior_findings: int) -> dict[str, object]:
        contradictions = self.detect(quotes)
        credibility = self.scorer.score(contradictions, prior_findings)
        return {
            "target": target,
            "credibility_score": credibility,
            "contradictions": contradictions,
            "attack_vectors": self._attack_vectors(contradictions),
        }

    def _attack_vectors(self, contradictions: list[Contradiction]) -> list[str]:
        vectors = []
        for item in contradictions[:5]:
            vectors.append(
                f"{item.inconsistency_axis}: {item.primary_quote_id} vs {item.impeaching_quote_id} ({item.severity})"
            )
        return vectors
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Iterable
import re

STANDARD_OF_REVIEW = {
    "custody": "great weight of the evidence / abuse of discretion / clear legal error",
    "disqualification": "abuse of discretion",
    "1983": "de novo",
    "ppo": "abuse of discretion",
    "appellate-procedure": "de novo",

@dataclass(slots=True)
class AuthorityNode:
    citation: str
    proposition: str
    jurisdiction: str
    mandatory: bool
    weight: int

@dataclass(slots=True)
class ResearchPackage:
    issue: str
    lane: str
    standard_of_review: str
    authorities: list[AuthorityNode] = field(default_factory=list)
    gaps: list[str] = field(default_factory=list)

class AuthorityAndPrecedentEngine:
    def __init__(self, authority_bank: Iterable[AuthorityNode]) -> None:
        self.authority_bank = list(authority_bank)

    def research(self, issue: str, lane: str) -> ResearchPackage:
        standard = self._determine_standard(issue, lane)
        authorities = self._match(issue, lane)
        gaps = self._gap_detect(issue, authorities)
        return ResearchPackage(issue=issue, lane=lane, standard_of_review=standard, authorities=authorities, gaps=gaps)

    def _determine_standard(self, issue: str, lane: str) -> str:
        haystack = f"{issue.lower()} {lane.lower()}"
        for key, value in STANDARD_OF_REVIEW.items():
            if key in haystack:
                return value
        return "mixed standard requiring issue-by-issue statement"

    def _match(self, issue: str, lane: str) -> list[AuthorityNode]:
        terms = set(re.findall(r"[a-z0-9.]+", issue.lower()))
        ranked: list[AuthorityNode] = []
        for node in self.authority_bank:
            text = f"{node.citation} {node.proposition}".lower()
            score = sum(1 for term in terms if term in text)
            if lane.lower() == "f" and "mcr 7.212" in text:
                score += 4
            if lane.lower() == "a" and "722.23" in text:
                score += 4
            if score:
                ranked.append(
                    AuthorityNode(
                        citation=node.citation,
                        proposition=node.proposition,
                        jurisdiction=node.jurisdiction,
                        mandatory=node.mandatory,
                        weight=node.weight + score,
                    )
                )
        return sorted(ranked, key=lambda node: (node.mandatory, node.weight), reverse=True)[:12]

    def _gap_detect(self, issue: str, authorities: list[AuthorityNode]) -> list[str]:
        gaps: list[str] = []
        citations = {node.citation for node in authorities}
        if "custody" in issue.lower() and "MCL 722.23" not in citations:
            gaps.append("Missing best-interest-factor anchor: MCL 722.23")
        if "appeal" in issue.lower() and "MCR 7.212" not in citations:
            gaps.append("Missing appellate brief formatting authority: MCR 7.212")
        if "1983" in issue.lower() and not any("1983" in node.citation.lower() for node in authorities):
            gaps.append("Missing federal civil-rights anchor for claim elements")
        if not authorities:
            gaps.append("No authority coverage; route to manual legal research immediately")
        return gaps

    def verify_citations(self, draft: str) -> list[str]:
        findings: list[str] = []
        for token in re.findall(r"(MCR\s+\d+\.\d+|MCL\s+\d+\.\d+|MRE\s+\d+)", draft):
            if not any(node.citation.startswith(token) for node in self.authority_bank):
                findings.append(f"Unverified citation detected: {token}")
        return findings
```

```python
from __future__ import annotations

from dataclasses import dataclass
import re

BANNED_TERMS = {
    "Jane Berry",
    "Patricia Berry",
    "P35878",
    "91% alienation",
    "Lincoln David Watson",
}

REQUIRED_IDENTITIES = {
    "child": "L.D.W.",
    "judge": "Hon. Jenny L. McNeill",
}

@dataclass(slots=True)
class GateResult:
    gate: str
    passed: bool
    detail: str

class QualityAssuranceFortress:
    def run(self, text: str) -> list[GateResult]:
        gates = [
            self._gate_child_privacy,
            self._gate_banned_terms,
            self._gate_case_numbers,
            self._gate_rule_format,
            self._gate_date_sanity,
            self._gate_caption_presence,
            self._gate_judge_identity,
            self._gate_citation_duplicates,
            self._gate_placeholder_sweep,
            self._gate_service_presence,
            self._gate_forum_language,
            self._gate_party_roles,
            self._gate_damages_numbers,
            self._gate_hallucination_markers,
            self._gate_traceability,
        ]
        return [gate(text) for gate in gates]

    def _gate_child_privacy(self, text: str) -> GateResult:
        banned_pattern = re.compile(r"\bL\.D\.W\.[A-Za-z]")
        return GateResult("child-privacy", not bool(banned_pattern.search(text)), "Child reference must remain L.D.W. only")

    def _gate_banned_terms(self, text: str) -> GateResult:
        found = [term for term in BANNED_TERMS if term in text]
        return GateResult("banned-terms", not found, f"Found banned terms: {found}" if found else "No banned terms detected")

    def _gate_case_numbers(self, text: str) -> GateResult:
        required = ["2024-001507-DC", "2025-002760-CZ", "2023-5907-PP", "366810"]
        present = [item for item in required if item in text]
        return GateResult("case-number-map", len(present) >= 1, f"Detected case references: {present}")

    def _gate_rule_format(self, text: str) -> GateResult:
        malformed = re.findall(r"MCR\s+\d{1,2}(?!\.\d)", text)
        return GateResult("rule-format", not malformed, f"Malformed MCR citations: {malformed}")

    def _gate_date_sanity(self, text: str) -> GateResult:
        years = {int(match) for match in re.findall(r"\b(20\d{2})\b", text)}
        valid = not years or all(2023 <= year <= 2026 for year in years)
        return GateResult("date-sanity", valid, f"Years detected: {sorted(years)}")

    def _gate_caption_presence(self, text: str) -> GateResult:
        required = "ANDREW JAMES PIGORS" in text and "EMILY A. WATSON" in text
        return GateResult("caption-presence", required, "Both principal parties must appear in captioned documents")

    def _gate_judge_identity(self, text: str) -> GateResult:
        okay = REQUIRED_IDENTITIES["judge"] in text or "COURT OF APPEALS" in text or "UNITED STATES DISTRICT COURT" in text
        return GateResult("judge-identity", okay, "Judge identity verified when state-court filing is detected")

    def _gate_citation_duplicates(self, text: str) -> GateResult:
        cites = re.findall(r"(MCR\s+\d+\.\d+|MCL\s+\d+\.\d+|MRE\s+\d+)", text)
        duplicates = sorted({cite for cite in cites if cites.count(cite) > 4})
        return GateResult("citation-duplicates", not duplicates, f"Overused citations: {duplicates}")

    def _gate_placeholder_sweep(self, text: str) -> GateResult:
        bad = [token for token in ("TODO", "TK", "[INSERT", "lorem ipsum") if token.lower() in text.lower()]
        return GateResult("placeholder-sweep", not bad, f"Placeholder tokens: {bad}")

    def _gate_service_presence(self, text: str) -> GateResult:
        okay = "CERTIFICATE OF SERVICE" in text or "service" in text.lower()
        return GateResult("service-presence", okay, "Service language is required in filing-ready output")

    def _gate_forum_language(self, text: str) -> GateResult:
        okay = any(token in text for token in ("STATE OF MICHIGAN", "COURT OF APPEALS", "UNITED STATES DISTRICT COURT"))
        return GateResult("forum-language", okay, "Forum header not detected")

    def _gate_party_roles(self, text: str) -> GateResult:
        okay = "Plaintiff" in text and ("Defendant" in text or "Defendants" in text)
        return GateResult("party-roles", okay, "Pleading should define party roles")

    def _gate_damages_numbers(self, text: str) -> GateResult:
        suspicious = re.findall(r"\$0{3,}", text)
        return GateResult("damages-numbers", not suspicious, f"Suspicious damages placeholders: {suspicious}")

    def _gate_hallucination_markers(self, text: str) -> GateResult:
        bad = [term for term in ("imaginary precedent", "fictional statute", "invented case") if term in text.lower()]
        return GateResult("hallucination-markers", not bad, f"Markers: {bad}")

    def _gate_traceability(self, text: str) -> GateResult:
        okay = any(token in text for token in ("Exhibit", "PIGORS-", "MCR ", "MCL "))
        return GateResult("traceability", okay, "Output must carry exhibits or authority anchors")

for result in fortress.run(sample_text):
    print(result.gate, result.passed, result.detail)
```

## AL8: Multi-Forum Submission Router

**Purpose**

Fuse jurisdiction analysis, e-filing compatibility, deadline enforcement, and forum-specific packaging rules into a final submission engine for MiFILE, TrueFiling, CM/ECF, or mail.

**Design Pattern**

Strategy Router + Deadline Buffering + Forum Contract Validation + Submission Scheduler

### Detailed Description

AL8 is the terminal action layer of APEX-LITIGATION. It takes validated packages and routes them to the correct forum according to court, lane, relief type, deadline, and packaging contract. Unlike a simple exporter, it understands that state trial filings, appellate submissions, federal pleadings, and misconduct packets each have different portal expectations and timing risks.

Its emergent behavior depends on the fact that every prior module has already shaped the payload. Because AL1 understands provenance, AL2 stabilizes exhibits, AL3 frames attacks, AL4 supplies authorities, AL5 drafts the text, AL6 builds the package, and AL7 clears quality gates, AL8 can make a true routing decision instead of guessing. The router is intelligent precisely because the full pipeline is integrated.

In the Pigors v. Each route has its own deadline buffer, artifact requirements, and release conditions. AL8 captures those differences while preserving a single operational dashboard.

The module's anti-hallucination role is understated but critical. It refuses to route incomplete packages, expired deadlines, or portal-incompatible artifacts. If the filing cannot lawfully or practically be sent, AL8 stops the chain rather than simulating completion.

### Code Example

```python
from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime, timedelta
from pathlib import Path

@dataclass(slots=True)
class SubmissionTarget:
    forum: str
    endpoint: str
    required_extensions: tuple[str, ...]
    deadline_buffer_hours: int

@dataclass(slots=True)
class SubmissionPackage:
    filing_id: str
    forum: str
    package_dir: str
    due_utc: datetime
    artifacts: list[str]

class MultiForumSubmissionRouter:
    TARGETS = {
        "MiFILE": SubmissionTarget("MiFILE", "https://mifile.courts.michigan.gov", (".pdf", ".json"), 6),
        "TrueFiling": SubmissionTarget("TrueFiling", "https://www.truefiling.com", (".pdf",), 12),
        "CM/ECF": SubmissionTarget("CM/ECF", "https://ecf.miwd.uscourts.gov", (".pdf",), 24),
        "MAIL": SubmissionTarget("MAIL", "physical-service", (".pdf", ".txt"), 48),
    }

    def route(self, package: SubmissionPackage) -> dict[str, object]:
        target = self.TARGETS[package.forum]
        self._validate(package, target)
        scheduled = package.due_utc - timedelta(hours=target.deadline_buffer_hours)
        return {
            "filing_id": package.filing_id,
            "forum": package.forum,
            "endpoint": target.endpoint,
            "scheduled_submission_utc": scheduled.isoformat(),
            "artifacts": package.artifacts,
            "package_dir": package.package_dir,
        }

    def _validate(self, package: SubmissionPackage, target: SubmissionTarget) -> None:
        package_dir = Path(package.package_dir)
        if not package_dir.exists():
            raise FileNotFoundError(package.package_dir)
        suffixes = {Path(path).suffix.lower() for path in package.artifacts}
        missing = [ext for ext in target.required_extensions if ext not in suffixes]
        if missing:
            raise ValueError(f"Missing required artifacts for {target.forum}: {missing}")
        if datetime.utcnow() >= package.due_utc:
            raise ValueError("Submission deadline already expired")
        if package.forum == "MAIL" and "certificate-of-service.txt" not in {Path(path).name for path in package.artifacts}:
            raise ValueError("Mail routing requires certificate-of-service.txt")

    def routing_note(self, forum: str) -> str:
        notes = {
            "MiFILE": "State-court package must preserve caption, judge assignment, and service metadata.",
            "TrueFiling": "Appellate package must include brief, appendix, and index of authorities.",
            "CM/ECF": "Federal package must include complaint, civil cover sheet, summons outputs, and exhibits.",
            "MAIL": "Physical routing reserved for JTC and selected MSC or agency submissions.",
        }
        return notes[forum]

router = MultiForumSubmissionRouter()
result = router.route(
    SubmissionPackage(
        filing_id="F-COA-2026-04-01",
        forum="TrueFiling",
        package_dir=r"C:\\LitigationOS\\filings\\F-COA-2026-04-01",
        due_utc=datetime.utcnow() + timedelta(days=2),
        artifacts=[
            r"C:\\LitigationOS\\filings\\F-COA-2026-04-01\\brief.pdf",
            r"C:\\LitigationOS\\filings\\F-COA-2026-04-01\\appendix.pdf",
        ],
    )
)
print(result["scheduled_submission_utc"])
print(router.routing_note("TrueFiling"))
```
