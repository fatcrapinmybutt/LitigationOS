## Module DB1: Schema Architect

**Purpose**: Accept a universal data model definition and decompose it into engine-specific
schemas — deciding which entities belong in which database based on access patterns, data
shape, relationship density, temporal characteristics, and vector dimensionality.

**Design Pattern**: Engine Selection Matrix → Schema Projection

### Engine Selection Matrix

```yaml
# Universal Data Model → Engine Routing Rules
engine_selection:
  postgresql:
    - ACID transactions required
    - Complex joins (>2 tables)
    - Strong consistency mandatory
    - Structured/tabular data
    - Financial or audit data
  mongodb:
    - Deeply nested documents
    - Schema-per-record flexibility
    - Read-heavy with embedded data
    - Rapid prototyping / schema evolution
  redis:
    - Sub-millisecond access required
    - Session/token storage
    - Rate limiting counters
    - Pub/sub event channels
    - Leaderboards / sorted sets
  neo4j:
    - Relationship-dense data (>3 hops)
    - Social graphs / recommendation engines
    - Dependency trees / knowledge graphs
    - Path-finding queries
  influxdb:
    - Time-stamped measurements
    - High write throughput (>100K/sec)
    - Downsampling / retention policies
    - IoT sensor data / metrics
  pgvector:
    - Embedding storage (768-3072 dims)
    - Similarity search / nearest neighbors
    - RAG retrieval backends
    - Semantic search indexes
```

### PostgreSQL Advanced Queries

```sql
-- Recursive CTE: Organization hierarchy with depth tracking
WITH RECURSIVE org_tree AS (
    -- Base case: root nodes
    SELECT id, name, parent_id, 1 AS depth,
           ARRAY[id] AS path,
           name::TEXT AS full_path
    FROM departments
    WHERE parent_id IS NULL
    
    UNION ALL
    
    -- Recursive step
    SELECT d.id, d.name, d.parent_id, ot.depth + 1,
           ot.path || d.id,
           ot.full_path || ' > ' || d.name
    FROM departments d
    INNER JOIN org_tree ot ON d.parent_id = ot.id
    WHERE NOT d.id = ANY(ot.path)  -- Cycle prevention
)
SELECT * FROM org_tree ORDER BY path;

-- Window functions: Running totals with partitioned ranking
SELECT
    order_date,
    customer_id,
    amount,
    SUM(amount) OVER (
        PARTITION BY customer_id
        ORDER BY order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_total,
    RANK() OVER (
        PARTITION BY customer_id
        ORDER BY amount DESC
    ) AS amount_rank,
    LAG(amount, 1) OVER (
        PARTITION BY customer_id
        ORDER BY order_date
    ) AS prev_amount,
    amount - LAG(amount, 1) OVER (
        PARTITION BY customer_id
        ORDER BY order_date
    ) AS amount_delta
FROM orders
WHERE order_date >= CURRENT_DATE - INTERVAL '90 days';

-- JSONB query with lateral join for flattening
SELECT p.id, p.name, attr.*
FROM products p,
LATERAL jsonb_to_record(p.attributes) AS attr(
    color TEXT,
    size TEXT,
    weight NUMERIC,
    tags TEXT[]
)
WHERE 'electronics' = ANY(p.categories)
  AND (p.attributes @> '{"featured": true}'
       OR p.price > 100);
```

### Neo4j Cypher Queries

```cypher
// Social graph: Friends-of-friends who purchased similar products
MATCH (user:User {id: $userId})-[:FOLLOWS]->(friend:User)
      -[:FOLLOWS]->(fof:User)-[:PURCHASED]->(product:Product)
WHERE NOT (user)-[:FOLLOWS]->(fof)
  AND NOT (user)-[:PURCHASED]->(product)
  AND fof <> user
WITH fof, product, COUNT(DISTINCT friend) AS mutual_connections
ORDER BY mutual_connections DESC
RETURN product.name AS recommendation,
       mutual_connections,
       COLLECT(DISTINCT fof.name)[..5] AS recommenders
LIMIT 10;

// Knowledge graph traversal with variable-length paths
MATCH path = (start:Concept {name: $concept})
              -[:RELATED_TO*1..4]-(end:Concept)
WHERE ALL(n IN nodes(path) WHERE n.confidence > 0.7)
WITH end, path,
     REDUCE(score = 1.0, r IN relationships(path) |
            score * r.weight) AS path_score
ORDER BY path_score DESC
RETURN end.name, path_score,
       [n IN nodes(path) | n.name] AS traversal
LIMIT 20;
```

### Zero-Downtime Migration Strategy

```sql
-- PHASE 1: EXPAND — Add new column (nullable, no breaking change)
-- Migration: 20260327_001_add_email_verified.sql
BEGIN;
ALTER TABLE users ADD COLUMN email_verified BOOLEAN;
-- No DEFAULT constraint = instant on PG 11+ (no table rewrite)
COMMENT ON COLUMN users.email_verified IS 'Added in migration 001 — backfill pending';
COMMIT;

-- PHASE 2: BACKFILL — Populate in batches (non-blocking)
-- Migration: 20260327_002_backfill_email_verified.sql
DO $$
DECLARE
    batch_size INT := 5000;
    rows_updated INT;
BEGIN
    LOOP
        UPDATE users
        SET email_verified = (email_confirmation_date IS NOT NULL)
        WHERE email_verified IS NULL
          AND id IN (
              SELECT id FROM users
              WHERE email_verified IS NULL
              ORDER BY id
              LIMIT batch_size
              FOR UPDATE SKIP LOCKED  -- Non-blocking!
          );
        
        GET DIAGNOSTICS rows_updated = ROW_COUNT;
        EXIT WHEN rows_updated = 0;
        
        RAISE NOTICE 'Backfilled % rows', rows_updated;
        PERFORM pg_sleep(0.1);  -- Throttle to reduce load
        COMMIT;
    END LOOP;
END $$;

-- PHASE 3: CONTRACT — Add constraint after backfill complete
-- Migration: 20260327_003_finalize_email_verified.sql
BEGIN;
ALTER TABLE users ALTER COLUMN email_verified SET NOT NULL;
ALTER TABLE users ALTER COLUMN email_verified SET DEFAULT false;
-- Drop old column if replaced
-- ALTER TABLE users DROP COLUMN email_confirmation_date;
COMMIT;
```

## Module DB4: Cache Strategist

**Purpose**: Design Redis/Memcached caching architectures including cache-aside, write-through,
write-behind patterns, invalidation strategies, cache warming, and distributed cache
coordination.

**Design Pattern**: Cache Pattern Selection → TTL Strategy → Invalidation Graph

### Redis Architecture Patterns

```python
import redis.asyncio as redis
from typing import Optional, Any
import json
import hashlib

class CacheStrategist:
    """DB4: Multi-pattern cache layer with automatic invalidation."""
    
    def __init__(self, redis_url: str):
        self.r = redis.from_url(redis_url, decode_responses=True)
    
    # ── Cache-Aside Pattern ──────────────────────────────────
    async def cache_aside(
        self, key: str, loader, ttl: int = 3600, stale_ttl: int = 86400
    ) -> Any:
        """Read-through with stale-while-revalidate."""
        cached = await self.r.get(f"cache:{key}")
        if cached:
            data = json.loads(cached)
            if data.get("_fresh"):
                return data["value"]
            # Stale: return immediately, refresh in background
            # (caller triggers async refresh)
            return data["value"]
        
        # Cache miss: load from source
        value = await loader()
        await self.r.setex(
            f"cache:{key}",
            stale_ttl,
            json.dumps({"value": value, "_fresh": True})
        )
        # Set a shorter TTL marker for freshness
        await self.r.setex(f"fresh:{key}", ttl, "1")
        return value
    
    # ── Write-Behind Pattern ─────────────────────────────────
    async def write_behind(self, entity: str, id: str, data: dict):
        """Buffer writes in Redis, flush to DB periodically."""
        pipe = self.r.pipeline()
        pipe.hset(f"wb:{entity}:{id}", mapping=data)
        pipe.sadd(f"wb:dirty:{entity}", id)
        pipe.expire(f"wb:{entity}:{id}", 3600)
        await pipe.execute()
    
    # ── Sorted Set Leaderboard ───────────────────────────────
    async def update_leaderboard(
        self, board: str, member: str, score: float
    ):
        """Real-time leaderboard with Redis sorted sets."""
        pipe = self.r.pipeline()
        pipe.zadd(f"lb:{board}", {member: score})
        pipe.zrevrank(f"lb:{board}", member)
        pipe.zcard(f"lb:{board}")
        results = await pipe.execute()
        return {"rank": results[1] + 1, "total": results[2]}
    
    async def get_leaderboard(
        self, board: str, start: int = 0, stop: int = 9
    ):
        """Top-N with scores."""
        return await self.r.zrevrange(
            f"lb:{board}", start, stop, withscores=True
        )
    
    # ── Pub/Sub Event Bus ────────────────────────────────────
    async def publish_invalidation(self, pattern: str, entity_id: str):
        """Broadcast cache invalidation across all app instances."""
        await self.r.publish(
            "cache:invalidate",
            json.dumps({"pattern": pattern, "id": entity_id})
        )
    
    # ── Rate Limiter (Sliding Window) ────────────────────────
    async def rate_limit(
        self, key: str, limit: int, window_sec: int
    ) -> tuple[bool, int]:
        """Sliding window rate limiter. Returns (allowed, remaining)."""
        now = int(asyncio.get_event_loop().time() * 1000)
        window_ms = window_sec * 1000
        pipe = self.r.pipeline()
        pipe.zremrangebyscore(f"rl:{key}", 0, now - window_ms)
        pipe.zadd(f"rl:{key}", {f"{now}": now})
        pipe.zcard(f"rl:{key}")
        pipe.expire(f"rl:{key}", window_sec)
        results = await pipe.execute()
        count = results[2]
        return (count <= limit, max(0, limit - count))
    
    # ── Redis Streams (Event Sourcing) ───────────────────────
    async def append_event(self, stream: str, event: dict):
        """Append to Redis Stream for event sourcing."""
        return await self.r.xadd(
            f"stream:{stream}",
            event,
            maxlen=100000,
            approximate=True
        )
    
    async def read_events(
        self, stream: str, last_id: str = "0", count: int = 100
    ):
        """Read events from stream after given ID."""
        return await self.r.xrange(
            f"stream:{stream}", min=last_id, count=count
        )
```

class GraphRelationalBridge:
    """Keep Neo4j graph and PostgreSQL in sync."""
    
    async def sync_user_to_graph(self, pg_pool, neo4j_driver, user_id: str):
        """Project relational user data into graph."""
        # Read from PostgreSQL
        async with pg_pool.acquire() as conn:
            user = await conn.fetchrow(

                user_id
            )
            purchases = await conn.fetch(

                   WHERE o.user_id = $1""",

                WITH u

                MERGE (u)-[r:PURCHASED]->(p)

### Hybrid Search (Vector + Full-Text + Filters)

```sql
-- DB6: Hybrid search combining vector similarity with keyword matching
-- Uses Reciprocal Rank Fusion (RRF) for score combination

WITH vector_results AS (
    SELECT id, content, metadata,
           1 - (embedding <=> $1::vector) AS vector_score,
           ROW_NUMBER() OVER (
               ORDER BY embedding <=> $1::vector
           ) AS vector_rank
    FROM document_embeddings
    WHERE source_type = $2
      AND metadata @> $3::jsonb           -- Pre-filter by metadata
    ORDER BY embedding <=> $1::vector
    LIMIT 50
),
text_results AS (
    SELECT id, content, metadata,
           ts_rank(
               to_tsvector('english', content),
               plainto_tsquery('english', $4)
           ) AS text_score,
           ROW_NUMBER() OVER (
               ORDER BY ts_rank(
                   to_tsvector('english', content),
                   plainto_tsquery('english', $4)
               ) DESC
           ) AS text_rank
    FROM document_embeddings
    WHERE source_type = $2
      AND to_tsvector('english', content) @@ plainto_tsquery('english', $4)
    LIMIT 50
),
rrf_combined AS (
    SELECT
        COALESCE(v.id, t.id) AS id,
        COALESCE(v.content, t.content) AS content,
        COALESCE(v.metadata, t.metadata) AS metadata,
        -- Reciprocal Rank Fusion: 1/(k + rank)
        COALESCE(1.0 / (60 + v.vector_rank), 0) AS v_rrf,
        COALESCE(1.0 / (60 + t.text_rank), 0) AS t_rrf
    FROM vector_results v
    FULL OUTER JOIN text_results t ON v.id = t.id
)
SELECT id, content, metadata,
       (0.7 * v_rrf + 0.3 * t_rrf) AS combined_score
FROM rrf_combined
ORDER BY combined_score DESC
LIMIT $5;
```

### Cross-Database Transaction (Saga)

```typescript
// DB8: Polyglot saga — coordinate writes across PostgreSQL + MongoDB + Redis + Neo4j
import { PolyglotSaga, SagaStep } from './saga-engine';

interface CreateOrderSaga {
  userId: string;
  productId: string;
  quantity: number;
}

const createOrderSaga = new PolyglotSaga<CreateOrderSaga>('create-order', [
  // Step 1: Reserve inventory in PostgreSQL (ACID)
  new SagaStep({
    name: 'reserve-inventory',
    engine: 'postgresql',
    async execute(ctx, pg) {
      const result = await pg.query(`
        UPDATE products
        SET stock = stock - $1
        WHERE id = $2 AND stock >= $1
        RETURNING id, stock, price
      `, [ctx.quantity, ctx.productId]);
      
      if (result.rowCount === 0) throw new Error('Insufficient stock');
      ctx.product = result.rows[0];
      ctx.amount = ctx.product.price * ctx.quantity;
    },
    async compensate(ctx, pg) {
      await pg.query(`
        UPDATE products SET stock = stock + $1 WHERE id = $2
      `, [ctx.quantity, ctx.productId]);
    }
  }),
  
  // Step 2: Create order document in MongoDB
  new SagaStep({
    name: 'create-order-doc',
    engine: 'mongodb',
    async execute(ctx, mongo) {
      const order = await mongo.collection('orders').insertOne({
        user_id: ctx.userId,
        product_id: ctx.productId,
        quantity: ctx.quantity,
        amount: ctx.amount,
        status: 'pending',
        created_at: new Date()
      });
      ctx.orderId = order.insertedId.toString();
    },
    async compensate(ctx, mongo) {
      await mongo.collection('orders').deleteOne({ _id: ctx.orderId });
    }
  }),
  
  // Step 3: Update social graph in Neo4j
  new SagaStep({
    name: 'update-purchase-graph',
    engine: 'neo4j',
    async execute(ctx, neo4j) {
      await neo4j.run(`
        MATCH (u:User {id: $userId}), (p:Product {id: $productId})
        MERGE (u)-[r:PURCHASED]->(p)
        SET r.last_date = datetime(),
            r.total_quantity = coalesce(r.total_quantity, 0) + $quantity,
            r.order_id = $orderId
      `, { userId: ctx.userId, productId: ctx.productId,
           quantity: ctx.quantity, orderId: ctx.orderId });
    },
    async compensate(ctx, neo4j) {
      await neo4j.run(`
        MATCH (u:User {id: $userId})-[r:PURCHASED]->(p:Product {id: $productId})
        WHERE r.order_id = $orderId
        DELETE r
      `, { userId: ctx.userId, productId: ctx.productId, orderId: ctx.orderId });
    }
  }),
  
  // Step 4: Invalidate caches in Redis
  new SagaStep({
    name: 'invalidate-caches',
    engine: 'redis',
    async execute(ctx, redis) {
      const pipe = redis.pipeline();
      pipe.del(`cache:product:${ctx.productId}`);
      pipe.del(`cache:user:${ctx.userId}:orders`);
      pipe.xadd('stream:orders', '*', {
        type: 'ORDER_CREATED',
        order_id: ctx.orderId,
        user_id: ctx.userId,
        amount: ctx.amount.toString()
      });
      await pipe.exec();
    },
    async compensate(ctx, redis) {
      // Cache invalidation is idempotent — no compensation needed
      await redis.xadd('stream:orders', '*', {
        type: 'ORDER_ROLLED_BACK', order_id: ctx.orderId
      });
    }
  })
]);

// Execute the saga
const result = await createOrderSaga.execute({
  userId: 'user-123',
  productId: 'prod-456',
  quantity: 2
});
// On any step failure, all previous steps are compensated in reverse order
```

### Federated Query Router

```typescript
// DB8: Route queries to optimal engine based on data location
class FederatedQueryRouter {
  private engines: Map<string, DatabaseEngine>;
  private entityRegistry: Map<string, string>; // entity → engine
  
  async query(request: FederatedQuery): Promise<any> {
    // Decompose query into engine-specific sub-queries
    const plan = this.planQuery(request);
    
    // Execute sub-queries in parallel where possible
    const results = await Promise.all(
      plan.steps.map(step => this.executeStep(step))
    );
    
    // Join results in application layer
    return this.joinResults(results, plan.joinStrategy);
  }
  
  private planQuery(request: FederatedQuery): QueryPlan {
    // Example: "Get user profile with purchase history,
    //           social connections, and product recommendations"
    return {
      steps: [
        { engine: 'postgresql', query: 'SELECT * FROM users WHERE id = $1',
          alias: 'user' },
        { engine: 'mongodb', query: { collection: 'orders', 
          filter: { user_id: '$userId' } }, alias: 'orders' },
        { engine: 'neo4j', query: `
          MATCH (u:User {id: $userId})-[:FOLLOWS]->(f:User)
          RETURN f.id, f.name LIMIT 10`,
          alias: 'connections' },
        { engine: 'redis', query: 'ZREVRANGE user:$userId:recommendations 0 4',
          alias: 'recommendations' }
      ],
      joinStrategy: {
        root: 'user',
        embed: {
          'orders': { on: 'user.id = orders.user_id', type: 'array' },
          'connections': { on: 'user.id', type: 'array' },
          'recommendations': { type: 'array' }
        }
      }
    };
  }
}
```
