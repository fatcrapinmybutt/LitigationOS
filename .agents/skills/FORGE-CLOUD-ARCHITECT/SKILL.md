### Provider Mapping Engine

```python
# Pseudo-code for the Cloud-Agnostic Modeler resolution engine

RESOURCE_MAP = {
    "container-orchestrated": {
        "aws":   {"service": "EKS",   "terraform": "aws_eks_cluster"},
        "azure": {"service": "AKS",   "terraform": "azurerm_kubernetes_cluster"},
        "gcp":   {"service": "GKE",   "terraform": "google_container_cluster"},
    },
    "relational:postgresql": {
        "aws":   {"service": "RDS",       "terraform": "aws_db_instance"},
        "azure": {"service": "Azure SQL", "terraform": "azurerm_postgresql_flexible_server"},
        "gcp":   {"service": "Cloud SQL", "terraform": "google_sql_database_instance"},
    },
    "object_store": {
        "aws":   {"service": "S3",     "terraform": "aws_s3_bucket"},
        "azure": {"service": "Blob",   "terraform": "azurerm_storage_account"},
        "gcp":   {"service": "GCS",    "terraform": "google_storage_bucket"},
    },
    "global_load_balancer": {
        "aws":   {"service": "CloudFront + ALB", "terraform": "aws_cloudfront_distribution"},
        "azure": {"service": "Front Door",       "terraform": "azurerm_frontdoor"},
        "gcp":   {"service": "Cloud Load Balancing", "terraform": "google_compute_global_forwarding_rule"},
    },
    "managed_identity": {
        "aws":   {"service": "Cognito",       "terraform": "aws_cognito_user_pool"},
        "azure": {"service": "AAD B2C",       "terraform": "azurerm_aadb2c_directory"},
        "gcp":   {"service": "Firebase Auth", "terraform": "google_firebase_project"},
    },
}

def resolve_intent(intent: dict, provider: str) -> list[Resource]:
    """Map cloud-agnostic intent to provider-specific resources."""
    resources = []
    for component, spec in intent.items():
        abstract_type = classify_component(component, spec)
        concrete = RESOURCE_MAP[abstract_type][provider]
        resources.append(Resource(
            provider=provider,
            service=concrete["service"],
            terraform_type=concrete["terraform"],
            config=translate_config(spec, provider),
        ))
    return resources
```

### CI/CD Pipeline for IaC

```yaml
# .github/workflows/terraform-deploy.yaml
name: Terraform Multi-Cloud Deploy

on:
  push:
    branches: [main]
    paths: ['infra/**']
  pull_request:
    paths: ['infra/**']

jobs:
  plan:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        provider: [aws, azure, gcp]
        environment: [staging, production]
    steps:
      - uses: actions/checkout@v4

      - uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: "1.7.0"

      - name: Terraform Init
        run: |
          cd infra/${{ matrix.provider }}
          terraform init -backend-config="key=${{ matrix.provider }}/${{ matrix.environment }}"

      - name: Terraform Plan
        run: |
          cd infra/${{ matrix.provider }}
          terraform workspace select ${{ matrix.environment }} || terraform workspace new ${{ matrix.environment }}
          terraform plan -out=tfplan -var="environment=${{ matrix.environment }}"

      - name: Cost Estimation
        uses: infracost/actions/setup@v3
      - run: infracost breakdown --path infra/${{ matrix.provider }} --format json > cost.json

  apply:
    needs: plan
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    environment: production
    steps:
      - name: Terraform Apply
        run: terraform apply -auto-approve tfplan
```

### Automated Failover Runbook

```bash
#!/usr/bin/env bash
# dr/failover-runbook.sh — Automated multi-region failover
set -euo pipefail

PRIMARY_REGION="us-east-1"
SECONDARY_REGION="eu-west-1"
CLUSTER_NAME="api-db-primary"
GLOBAL_CLUSTER="global-api-db"

echo "=== DISASTER RECOVERY FAILOVER INITIATED ==="
echo "Timestamp: $(date -u +%Y-%m-%dT%H:%M:%SZ)"
echo "Primary:   ${PRIMARY_REGION}"
echo "Secondary: ${SECONDARY_REGION}"

# Step 1: Verify primary is unreachable
echo "[1/6] Verifying primary region health..."
if aws rds describe-db-clusters \
  --region "$PRIMARY_REGION" \
  --db-cluster-identifier "$CLUSTER_NAME" \
  --query 'DBClusters[0].Status' \
  --output text 2>/dev/null | grep -q "available"; then
  echo "WARNING: Primary appears healthy. Confirm failover intent."
  read -p "Continue? (yes/no): " confirm
  [[ "$confirm" == "yes" ]] || exit 1
fi

# Step 2: Promote secondary database
echo "[2/6] Promoting secondary Aurora cluster..."
aws rds failover-global-cluster \
  --region "$SECONDARY_REGION" \
  --global-cluster-identifier "$GLOBAL_CLUSTER" \
  --target-db-cluster-identifier \
    "arn:aws:rds:${SECONDARY_REGION}:${ACCOUNT_ID}:cluster:api-db-secondary"

# Step 3: Update DNS to point to secondary
echo "[3/6] Updating Route 53 DNS records..."
aws route53 change-resource-record-sets \
  --hosted-zone-id "$HOSTED_ZONE_ID" \
  --change-batch '{
    "Changes": [{
      "Action": "UPSERT",
      "ResourceRecordSet": {
        "Name": "api.example.com",
        "Type": "A",
        "AliasTarget": {
          "HostedZoneId": "'$SECONDARY_ALB_ZONE'",
          "DNSName": "'$SECONDARY_ALB_DNS'",
          "EvaluateTargetHealth": true
        }
      }
    }]
  }'

# Step 4: Scale up secondary compute
echo "[4/6] Scaling secondary EKS node group..."
aws eks update-nodegroup-config \
  --region "$SECONDARY_REGION" \
  --cluster-name "api-cluster-secondary" \
  --nodegroup-name "workers" \
  --scaling-config minSize=5,maxSize=50,desiredSize=10

# Step 5: Invalidate CDN cache
echo "[5/6] Invalidating CloudFront cache..."
aws cloudfront create-invalidation \
  --distribution-id "$CF_DISTRIBUTION_ID" \
  --paths "/*"

# Step 6: Verify
echo "[6/6] Verifying secondary region services..."
for i in {1..30}; do
  HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" "https://api.example.com/healthz")
  if [[ "$HTTP_CODE" == "200" ]]; then
    echo "✅ Failover complete. API responding 200 from ${SECONDARY_REGION}"
    exit 0
  fi
  echo "  Waiting... (attempt $i/30, got HTTP $HTTP_CODE)"
  sleep 10
done

echo "❌ Failover verification FAILED after 5 minutes"
exit 1
```

### Architecture Decision Shortcuts

| Decision | Choose A | Choose B |
|---|---|---|
| Containers vs Serverless | Long-running, stateful, >15min | Short, event-driven, <15min |
| Managed K8s vs Self-hosted | Always managed (EKS/AKS/GKE) | Never self-host in cloud |
| Multi-region vs Single | RTO < 1hr or regulatory | Cost-sensitive, RTO > 4hr |
| Reserved vs On-Demand | Steady-state base load | Variable/spiky workloads |
| SQL vs NoSQL | ACID, complex queries, joins | Key-value, scale-out, flexible schema |
| Monorepo IaC vs Multi-repo | Small team, < 50 resources | Large team, > 200 resources |
